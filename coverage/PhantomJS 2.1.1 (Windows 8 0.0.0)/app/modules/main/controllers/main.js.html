        function getCategories()
            {
                var deferred = $q.defer();
                $cache.getJobCategory(
                {
                    lang: 'vi'
                }, function(result)
                {
                    if (result.status)
                    {
                        $scope.listCategories = result.data.categoryList;
                        deferred.resolve(true);
                    }
                    else
                    {
                        deferred.resolve(false);
                    }
                });
                return deferred.promise;
            }
 
        function getPositions()
            {
                var deferred = $q.defer();
                $cache.getJobPosition(
                {
                    lang: 'vi'
                }, function(result)
                {
                    if (result.status)
                    {
                        $scope.listPosition = result.data.positionList;
                        deferred.resolve(true);

                    else
                    {
                        deferred.resolve(false);
                    }
                });
                return deferred.promise;
            }
            
        function getLocations()
        {
            var deferred = $q.defer();
            $cache.getJobLocation(
            {
                lang: 'vi'
            }, function(result)
            {
                if (result.status)
                {
                    result.data.countryList.unshift(
                    {
                        id: '',
                        title: '---' + allcountry + '---'
                    });
                    $scope.listLocation = result.data.countryList;
                    $scope.listProvince = result.data.provinceList;
                    deferred.resolve(true);
                }
                else
                {
                    deferred.resolve(false);
                }
            });
            return deferred.promise;
        }

            $common.searchJob($scope.search, function(result) {

        $scope.searchJob = function()
        {
            $scope.search.option.countryId = $scope.countries.selectedCountry.id;
            $scope.search.option.provinceId = $scope.provinces.selectedProvince.id;
            var info = $employee.setSearchOptions($scope.search);
            $common.searchJob(info, function(result)
            {
                if (result.status)
                {
                    var jobList = result.data.jobList;
                    if (!Array.isArray(jobList) || jobList.length === 0)
                    {
                        $scope.jobList = [];
                        $scope.showNotication = true;
                    }
                    else
                    {
                        localStorageService.set('jobList', jobList);
                        $scope.showNotication = false;
                        jobList = $filter('orderBy')(jobList, 'id', true);
                        jobList.map(function(job)
                        {
                            // add provinde name
                            var existProvince = _.find($scope.listProvince, function(
                                province)
                            {
                                return province.id == job.provinceId;
                            });
                            job.provinceName = existProvince ? existProvince.title :
                                '';
                            // add country name
                            var existLocation = _.find($scope.listLocation, function(
                                location)
                            {
                                return location.id == job.countryId;
                            })
                            job.countryName = existLocation ? existLocation.title :
                                '';
                        });
 
                        localStorageService.set('jobList', jobList);
                        $scope.jobListPagination = jobList.slice(0, $scope.itemPerPage);
                        $scope.jobList = jobList;
                        $scope.totalJobs = $scope.jobList.length;
                        $scope.totalPages = Math.ceil($scope.totalJobs / $scope.itemPerPage);
                        $scope.currPage = 1;
                        $scope.maxSize = 4;
                    }
                }
                else
                {
                    $scope.showNotication = true;
                    $scope.jobList = [];
                }
                $scope.jobListInit = [];
            });
        }
