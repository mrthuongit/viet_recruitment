    var _createInterviewQuestion = function(listQuestion, interviewId, callback)
    {
        if (listQuestion.length == 0)
        {
            callback(true);
        }
        else
        {
            var info = {
                token: $scope.userEmployer.token,
                interviewId: interviewId,
                questionList: listQuestion
            };
            $employer.addInterviewQuestion(info, function(result)
            {
                if (result.status)
                {
                    callback(true);
                }
                else
                {
                    callback(false);
                }
            });
        }
    };
    var _updateInterviewQuestion = function(listQuestion, interviewId, callback)
    {
        if (listQuestion.length == 0)
        {
            callback(true);
        }
        else
        {
            var info = {
                token: $scope.userEmployer.token,
                interviewId: interviewId,
                questionList: listQuestion
            };
            $employer.updateInterviewQuestion(info, function(result)
            {
                if (result.status)
                {
                    callback(true);
                }
                else
                {
                    callback(false);
                }
            });
        }
    };
    var _removeInterviewQuestion = function(listQuestion, interviewId, callback)
    {
        if (listQuestion.length == 0)
        {
            callback(true);
        }
        else
        {
            var questionIds = [];
            listQuestion.forEach(function(question)
            {
                questionIds.push(question.id);
            })
            var info = {
                token: $scope.userEmployer.token,
                questionIds: questionIds
            };
            $employer.removeInterviewQuestion(info, function(result)
            {
                if (result.status)
                {
                    callback(true);
                }
                else
                {
                    callback(false);
                }
            });
        }
    };
    var _createOrUpdateInterview = function(callback)
    {
        if ($scope.interview.id)
        {
            $employer.updateInterview(
            {
                token: $scope.userEmployer.token,
                interview: $scope.interview
            }, function(result)
            {
                callback($scope.interview.id);
            })
        }
        else
        {
            $employer.createInterview(
            {
                token: $scope.userEmployer.token,
                assignmentId: chooseAssignment.id,
                interview: $scope.interview
            }, function(result)
            {
                if (result.status)
                {
                    callback(result.data.interviewId);
                }
                else
                {
                    callback(false);
                }
            });
        }
    };
    var _createOrUpdateInterviewQuestion = function(interviewId, callback)
    {
        var createInterviewQuestion = _.filter($scope.interviewQuestionList, function(question)
        {
            return !question.id;
        });
        var updateInterviewQuestion = _.filter($scope.interviewQuestionList, function(question)
        {
            return question.id && !question.remove;
        });
        var removeInterviewQuestion = _.filter($scope.interviewQuestionList, function(question)
        {
            return question.id && question.remove;
        });
        _createInterviewQuestion(createInterviewQuestion, interviewId, function(createResult)
        {
            if (createResult)
            {
                _updateInterviewQuestion(updateInterviewQuestion, interviewId, function(
                    updateResult)
                {
                    if (updateResult)
                    {
                        _removeInterviewQuestion(removeInterviewQuestion,
                            interviewId,
                            function(removeResult)
                            {
                                if (removeResult)
                                {
                                    callback(true);
                                }
                                else
                                {
                                    callback(false);
                                }
                            });
                    }
                    else
                    {
                        callback(false);
                    }
                });
            }
            else
            {
                callback(false);
            }
        });
    };
    $scope.doCancel = function()
    {
        $location.path("/employer/interview_list");
    }
    $scope.doSave = function()
    {
        _createOrUpdateInterview(function(interviewId)
        {
            if (interviewId)
            {
                _createOrUpdateInterviewQuestion(interviewId, function(reuslt)
                {
                    localStorageService.remove("interviewQuestionList");
                    localStorageService.remove("interview");
                    $location.path("/employer/interview_list");
                });
            }
        });
    }
    init();
});
