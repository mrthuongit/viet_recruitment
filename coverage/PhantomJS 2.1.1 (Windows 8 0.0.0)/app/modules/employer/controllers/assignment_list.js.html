angular.module('employerModule').controller('AssignmentListController', function($scope, $employer,
    $translate, $rootScope,  localStorageService, $location, $q)
{
    var alert = $translate.instant('toarster.alert');
    var unablejob = $translate.instant('toaster.unablejob');
    $scope.goToDetailAssign = function(assign)
    {
        localStorageService.set('chooseAssignment', assign);
        localStorageService.remove("interviewQuestionList");
        localStorageService.remove('interview');
        $location.path("/employer/assignment");
    };
    $scope.deleteAssign = function(assign)
    {
        if (assign.status != 'initial') toaster.error( alert, unablejob);
        else
        {
            $employer.deleteAssignment(
            {
                token: $scope.userEmployer.token,
                assignmentId: assign.id
            }, function(result)
            {
                if (result.status)
                {}
                getAssignment();
            });
        }
    };
    $scope.goToAssignment = function()
    {
        $location.path("/employer");
    };
    $scope.newAssignment = function()
    {
        localStorageService.remove('chooseAssignment');
        localStorageService.remove("interviewQuestionList");
        localStorageService.remove('interview');
        $location.path("/employer/assignment");
    }
    $scope.setPage = function(pageNo)
    {
        $scope.currentPage = pageNo;
    };
    $scope.pageChanged = function()
    {
        $log.info('Page changed to: ' + $scope.currentPage);
    };
    $scope.setItemsPerPage = function(num)
    {
        $scope.itemsPerPage = num;
        $scope.currentPage = 1; //reset to first paghe
    }
    $scope.setSearch = function(search)
    {
        $scope.totalItems = $scope.list.length;
    };
    var getAssignment = function()
    {
        var deffered = $q.defer();
        var info = {
            token: $scope.userEmployer.token
        };
        $employer.getAssignment(info, function(result)
        {
            if (result.status)
            {
                $scope.listAssignment = result.data.assignmentList;
                localStorageService.set("listAssignment", $scope.listAssignment);
                var num_length = $scope.listAssignment.length;
                for (var i = 0; i < $scope.listAssignment.length; i++)
                {
                    $scope.listAssignment[i].index = num_length;
                    num_length = num_length - 1;
                }
                $scope.totalItems = $scope.listAssignment.length;
                $scope.currentPage = 1;
                $scope.itemsPerPage = $scope.viewby;
                $scope.maxSize = 5;
            }
            deffered.resolve();
        });
        return deffered.promise;
    }
    var init = function()
    {
        localStorageService.set("page", "assignment");
        $scope.userEmployer = localStorageService.get('userEmployer');
        if (!$scope.userEmployer)
        {
            $location.path("/");
            return;
        }
        $scope.listAssignment = [];
        getAssignment();
    }
    init();
});
