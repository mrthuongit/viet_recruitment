angular.module('employerModule').controller('QuestionConferenceInterviewController', function($scope, $employer, $translate, $rootScope, toaster, localStorageService, $location, $common, $q) {
    var alert = $translate.instant('toarster.alert');
    var required = $translate.instant('toarster.required');
    var noQuestion = $translate.instant('toarster.noQuestion');
      
    var questionList = [];
    var getInterviewQuestion = function(defer) {
        if (localStorageService.get("interviewQuestionList")) {
            $scope.interviewQuestionList = localStorageService.get("interviewQuestionList");
            defer.resolve();
        } else {
            if ($scope.interview.id) {
                $employer.getInterviewQuestion({
                    token: $scope.userEmployer.token,
                    interviewId: $scope.interview.id
                }, function(result) {
                    if (result.status) {
                        $scope.interviewQuestionList = result.data.questionList;
                        localStorageService.set("interviewQuestionList", $scope.interviewQuestionList);
                    }
                    defer.resolve();
                   
                })
            } else {
                defer.resolve();
            }
        }
    }
    var getQuestionCategory = function(lang, defer) {
        if (!localStorageService.get("questionCategory_" + lang)) {
            $common.getQuestionCategory({
                token: $scope.userEmployer.token,
                lang:lang
            }, function(result) {
                if (result.status) {
                    localStorageService.set("questionCategory_" + lang, result.data.categoryList);
                    $scope.questionCategory = result.data.categoryList;
                }
                defer.resolve();
            });
        } else {
            $scope.questionCategory = localStorageService.get("questionCategory_" + lang);
            defer.resolve();
        }
    }
    var getQuestion = function(lang, defer) {
        if (!localStorageService.get("questionList_" + lang)) {
        	$common.getQuestion({
                token: $scope.userEmployer.token,
                lang:lang
            }, function(result) {
                if (result.status) {
                    // $log.info(result.data.questionList);
                    localStorageService.set("questionList_" + lang, result.data.questionList);
                }
                defer.resolve();
            });
        } else defer.resolve();
    }
    
    var init = function() {
        $scope.interview = localStorageService.get('interview');
        $scope.userEmployer = localStorageService.get('userEmployer');
        if (!$rootScope.language) {
            $rootScope.language = 'vi';
            localStorageService.set('language',$rootScope.language);
        } 
        $scope.questionCategory = [];
        $scope.interviewQuestion = {
            questionFile: null
        };
 
        $scope.videoControl = null;
        $scope.interviewQuestionList = [];
        $scope.questionFile = null;
        $scope.questionUrl = null;
        
        var deferArray = [$q.defer(), $q.defer(), $q.defer(), $q.defer(), $q.defer()];
        var loopPromises = [];
        angular.forEach(deferArray, function(deferItem) {
            loopPromises.push(deferItem.promise);
        });
        
        deferArray[0].promise.then(function() {
            getQuestion('en', deferArray[1]);
        });
        deferArray[1].promise.then(function() {
            getQuestion('vi', deferArray[2]);
        });
        deferArray[2].promise.then(function() {
            getQuestionCategory('en', deferArray[3]);
        });
        deferArray[3].promise.then(function() {
            getQuestionCategory('vi', deferArray[4]);
        });
        getInterviewQuestion(deferArray[0]);
        $q.all(loopPromises).then(function() {
        	$scope.questionCategory = localStorageService.get("questionCategory_" + $rootScope.language);
            questionList = localStorageService.get("questionList_" + $rootScope.language);
            if ($scope.questionCategory.length > 0) {
                if (questionList.length > 0) {
                    $scope.questionList = _.filter(questionList, function(question) {
                        return question.categoryId === $scope.questionCategory[0].id;
                    });
                    $scope.interviewQuestion.categoryId = $scope.questionCategory[0].id;
                    $scope.interviewQuestion.questionId = $scope.questionList[0].id;
                }
            }
        });
    };
    
    $rootScope.$watch('language', function(newValue, oldValue) {
    	if (localStorageService.get("questionCategory_" + newValue)) {
            $scope.questionCategory = localStorageService.get("questionCategory_" + newValue); 
            $scope.interviewQuestion.categoryId = $scope.questionCategory[0].id;
        }
    	if (localStorageService.get("questionList_" + newValue)) {
    		questionList = localStorageService.get("questionList_" + newValue);
    		$scope.questionList = _.filter(questionList, function(question) {
                return question.categoryId === $scope.questionCategory[0].id;
            });
            $scope.interviewQuestion.questionId = $scope.questionList[0].id;
        }
    });
    
    
    $scope.selectCategory = function() {
        var lang = $rootScope.language;
        questionList = localStorageService.get("questionList_" + lang);
        $scope.questionList = _.filter(questionList, function(question) {
            return question.categoryId === $scope.interviewQuestion.categoryId;
        });
        $scope.interviewQuestion.questionId = $scope.questionList[0].id;
    };
    $scope.chooseQuestion = function() {
        var chooseQuestion = _.find($scope.questionList, function(question) {
            return question.id === $scope.interviewQuestion.questionId;
        });
        _addQuestionSystem(chooseQuestion);
    };
    var _addQuestionSystem = function(chooseQuestion) {
        var existQuestion = _.find($scope.interviewQuestionList, function(question) {
            return question.questionId === chooseQuestion.id;
        });
        if (!existQuestion) {
            var interviewQuestion = {
                questionId: chooseQuestion.id,
                id: null,
                order: $scope.interviewQuestionList.length + 1,
                source: "system",
                type: "text",
                response: $scope.interview.response,
                prepare: $scope.interview.prepare,
                title: chooseQuestion.title,
                videoUrl: ""
            }
            $scope.interviewQuestionList.push(interviewQuestion);
            localStorageService.set("interviewQuestionList", $scope.interviewQuestionList);
        }
    };
    $scope.removeInterviewQuestion = function(interviewQuestion) {
        if (interviewQuestion.id) 
        	interviewQuestion.remove = true;
        else
        	$scope.interviewQuestionList = _.reject($scope.interviewQuestionList, function(question) {
                    return question.title === interviewQuestion.title;
                
            });
        var order =1;
        for (var i=0;i<$scope.interviewQuestionList.length;i++ ) 
        	if (!$scope.interviewQuestionList[i].remove)
        		$scope.interviewQuestionList[i].order = order++;        
        
        localStorageService.set("interviewQuestionList", $scope.interviewQuestionList);
        
    };
    $scope.moveDownInterviewQuestion = function(interviewQuestion) {
        var existQuestion = _.find($scope.interviewQuestionList, function(question) {
            return  !question.remove && question.order > interviewQuestion.order;
        });
        if (existQuestion) {
        	var order = existQuestion.order;
        	existQuestion.order = interviewQuestion.order;
        	interviewQuestion.order = order;
        }
    };
    $scope.chooseQuestionSearch = function(question) {
        if (question && typeof question.originalObject === 'object') {
            _addQuestionSystem(question.originalObject);
        }
    };
    $scope.addQuestionManual = function() {
    	var questionCount = _.countBy($scope.interviewQuestionList, function(question) {
    		  return !question.remove;
    	});
        if(!questionCount.true) {
            questionCount = {
                true : 0
            }
        }
        if ($scope.interviewQuestion.manualQuestion.length > 0) {
            if ($scope.interviewQuestion.questionFile) {
                $common.uploadVideo({
                    file: $scope.interviewQuestion.questionFile
                }, function(result) {
                    if (result.status && result.data.result) {
                        var interviewQuestion = {
                            id: null,
                            order: questionCount.true + 1,
                            source: "manual",
                            type: "video",
                            response: $scope.interview.response,
                            prepare: $scope.interview.prepare,
                            title: $scope.interviewQuestion.manualQuestion,
                            videoUrl: result.data.url
                        }
                            $scope.interviewQuestionList.push(interviewQuestion);
                        $scope.interviewQuestion.manualQuestion = "";
                        $scope.interviewQuestion.questionFile = null;
                        $scope.questionUrl = null;
                        localStorageService.set("interviewQuestionList", $scope.interviewQuestionList);
                        $scope.$apply();
                    }
                });
            } else {
                var interviewQuestion = {
                    id: null,
                    order: questionCount.true + 1,
                    source: "manual",
                    type: "text",
                    response: $scope.interview.response,
                    prepare: $scope.interview.prepare,
                    title: $scope.interviewQuestion.manualQuestion,
                    videoUrl: null,
                }
                    $scope.interviewQuestionList.push(interviewQuestion);
                $scope.interviewQuestion.manualQuestion = "";
                localStorageService.set("interviewQuestionList", $scope.interviewQuestionList);
            }
        }
    };
    
    $scope.doCancel = function() {
    	$location.path("/employer/interview_list");
    }
    
    
   
    $scope.backSetting = function() {
        localStorageService.set('interview', $scope.interview);
        $location.path("/employer/conference/interview");
    };
    $scope.gotoReview = function() {
            localStorageService.set('interview', $scope.interview);
            localStorageService.set("interviewQuestionList", $scope.interviewQuestionList);
            $location.path("/employer/conference/review_confirm_interview");
       
    };
    init();
});
