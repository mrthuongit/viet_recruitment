angular.module('employerModule')
.controller('CandidateApplyListController', function($scope, $employer, $translate, $rootScope, $filter, $window, toaster, localStorageService, $location, $q) {
    localStorageService.set('page', 'candidate_apply_list');
    $scope.userEmployer = localStorageService.get('userEmployer');
    // Pagination
    $scope.candidateApplyList = [];
    $scope.candidateApplyListPagination = [];
	$scope.totalCandidates = 0;
	$scope.currPage = 1;
	$scope.itemsPerPage = 10;
	$scope.maxSize = 4;
    
    var getCandidateApplyList = function() {
        var deffered = $q.defer();
        var info = { token: $scope.userEmployer.token };
 
        $employer.getCandidateApplyList(info, function(result) {
            if (result.status) {
                deffered.resolve(result.data.candidateList);
            } else {
                $log.info('Khong lay duoc license');
                deffered.resolve(null);
            }            
        });
 
        return deffered.promise;
    }
 
    var calcPagination = function(currPage, listItems) {
    	var start = (currPage - 1)*$scope.itemsPerPage;
      	var end = currPage*$scope.itemsPerPage;
 
      	$scope.currPage = currPage;
     	$scope.candidateApplyListPagination = listItems.slice(start, end);
    }  
 
    var getFullname = function(candidatesList) {
    	candidatesList.map(function(candidate) {
 
    		if (candidate.name.indexOf('-') !== -1) {
    			var firstname = candidate.name.split('-')[0];
    			var lastname = candidate.name.split('-')[1];
    			candidate.fullname = firstname + ' ' + lastname;
    		} else {
    			candidate.fullname = candidate.name;
    		}
 
    		return candidate;
    	});
    }  
 
    getCandidateApplyList()
    .then(function(candidateApplyList) {
        if (candidateApplyList) {
        	getFullname(candidateApplyList);
            $scope.candidateApplyList = $filter('orderBy')(candidateApplyList, 'id', true);
        	$scope.currCandidateApplyList = $scope.candidateApplyList;
        	$scope.totalCandidates = $scope.currCandidateApplyList.length;
        	calcPagination(1, $scope.currCandidateApplyList);
        }
    });
 
    $scope.changePage = function(currPage) {
    	$window.scrollTo(0, 0);
        calcPagination(currPage, $scope.currCandidateApplyList);
    }
 
    $scope.changeItemsPerPage = function() {
        $scope.itemsPerPage = $scope.optionNumbers.selectedNumber.value;
        $scope.numPages = Math.ceil($scope.candidateApplyList/$scope.itemsPerPage);
 
        if ($scope.currPage > $scope.totalPages) {
            $scope.currPage = $scope.totalPages;
        } 
 
        var start = ($scope.currPage - 1)*$scope.itemsPerPage;
        var end = $scope.currPage*$scope.itemsPerPage;
        $scope.candidateApplyListPagination = $scope.candidateApplyList.slice(start, end);
    }
 
    $scope.filterCandidateApplyList = function(filterText) {
    	if (filterText !== '') {
    		$scope.currCandidateApplyList = $filter('filter')($scope.candidateApplyList, { jobName: filterText });
    		$scope.totalCandidates = $scope.currCandidateApplyList.length;
    		calcPagination(1, $scope.currCandidateApplyList);
    	} else {
    		$scope.currCandidateApplyList = $scope.candidateApplyList;
    		$scope.totalCandidates = $scope.currCandidateApplyList.length;
    		calcPagination(1, $scope.currCandidateApplyList);
    	}
    }
});
