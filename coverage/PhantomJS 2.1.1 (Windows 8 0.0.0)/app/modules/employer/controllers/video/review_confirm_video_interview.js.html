    var getvideoProfile = function() {
        if(!localStorageService.get('companyProfile')) {
            $employer.getCompanyEmployer({token: $scope.userEmployer.token}, function(result){
              if(result.status){
                $scope.company = result.data.company;
                $scope.interview.aboutUsUrl = $scope.company.videoUrl;
                localStorageService.set('companyProfile', $scope.company);
              }
            });
        } else {
            $scope.company = localStorageService.get('companyProfile');
            $scope.interview.aboutUsUrl = $scope.company.videoUrl;
        }
    }

      if(listQuestion.length == 0){

    var _createOrUpdateInterview = function(callback){
      if($scope.interview.id){
        $employer.updateInterview({token: $scope.userEmployer.token,interview: $scope.interview}, function(result){
          callback($scope.interview.id);
        })

        $employer.createInterview({token:$scope.userEmployer.token, assignmentId:chooseAssignment.id, interview: $scope.interview}, function(result){
          if(result.status && result.data.result){
            callback(result.data.interviewId);

            callback(false);
          }
        });
      }
    };
    var _createOrUpdateInterviewQuestion = function(interviewId, callback){
      var createInterviewQuestion = _.filter($scope.interviewQuestionList, function(question){
        return !question.id;
      });
      var updateInterviewQuestion = _.filter($scope.interviewQuestionList, function(question){
        return question.id && !question.remove;
      });
      var removeInterviewQuestion = _.filter($scope.interviewQuestionList, function(question){
        return question.id && question.remove;
      });
      
      _createInterviewQuestion(createInterviewQuestion, interviewId, function(createResult){
        if(createResult){
          _updateInterviewQuestion(updateInterviewQuestion, interviewId, function(updateResult){
            if(updateResult){
              _removeInterviewQuestion(removeInterviewQuestion, interviewId, function(removeResult){
                if(removeResult){
                  callback(true);
                } else{
                  callback(false);
                }
              });
            } else{
              callback(false);
            }
          });
        } else{
          callback(false);
        }
      });
 
    };
