angular.module('employerModule').controller('EmployerMenuController', function($scope, $employer, $translate,
    $rootScope, $q, localStorageService, $location, blockUI)
{
    $scope.licenseCategory = null;
    $scope.userEmployer = localStorageService.get('userEmployer');
    $scope.goLogout = function()
    {
        blockUI.start();
        $employer.logout(
        {
            token: $scope.userEmployer.token
        }, function(result)
        {
            if (result.status)
            {
                $location.path("/");
                localStorageService.clearAll();
            }
            blockUI.stop();
        });
    };
    $scope.userEmployer = localStorageService.get("userEmployer");
    if (!$scope.userEmployer)
    {
        $location.path("/");
        return;
    }
    $scope.page = localStorageService.get("page");
    var getLicenseEmployer = function()
    {
        var deffered = $q.defer();
        var info = {
            token: $scope.userEmployer.token
        };
        $employer.getLicenseInfo(info, function(result)
        {
            if (result.status)
            {
                deffered.resolve(result.data.licenseInfo);
            }
            else
            {
                $log.info('Khong lay duoc license');
                deffered.resolve(null);
            }
        });
        return deffered.promise;
    }
    getLicenseEmployer().then(function(licenseInfo)
    {
        if (licenseInfo)
        {
            $scope.licenseCategory = licenseInfo.license.code.toLowerCase();
        }
    });
});
