'use strict';
angular.module('customerCareModule').controller('CustomerCareController', function($scope, $rootScope,
    $location, $routeParams, $admin, _, localStorageService, blockUI)
{
    $rootScope.userCustomerCare = localStorageService.get('userCustomerCare');
    $rootScope.viewCustomerCare = {
        page: "company"
    };
    $rootScope.language = 'vi';
    $scope.doLogout = function()
    {
        $admin.logoutAccount(
        {
            token: $rootScope.userCustomerCare.token
        }, function(result)
        {
            if (result.status)
            {
                $scope.userCustomerCare = null;
                localStorageService.remove('userCustomerCare');
                localStorageService.remove('chooseAssignmentCustomerCare');
                localStorageService.remove('companyList');
                localStorageService.remove('listAssignmentCustomerCare');
                $location.path('/cc');
            }
        });
    }
});
